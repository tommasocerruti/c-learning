#define NUM 9

//using dynamic memory allocation
//idea: fixed size array[9]=[(1,0),(2,0),...,(9,0)]

    int num, occ;
}val;

        fprintf(stderr,"Error file opening");
        exit(1);
    }
    for (i=0;i<NUM;i++){
        vals[i].num=i+1;

    }


};
